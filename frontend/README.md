## 폴더 구조

```
  root
    ├── public
    ├── src
    │     ├── apis                    # API 관련
    │     ├── assets                  # 미디어 파일들
    │     │      ├── images           # 이미지 파일 관련
    │     │      └── styles           # 폰트, 색상
    │     │             └── fonts
    │     ├── components              # 컴포넌트
    │     │      └── common           # (공통,페이지별)
    │     ├── routers                 # 라우팅
    │     ├── utils                   # 유틸함수
    │     ├── screens                 # 페이지별
    │     └── store                   # 상태관리
    └── dist                          # 빌드 파일
          └── ...
```
